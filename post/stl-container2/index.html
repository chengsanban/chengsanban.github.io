#include<iostream>
#include<set>
using namespace std;
int main()
{ 
    set<int> s1; 
    s1.insert(1); 
    s1.insert(45); 
    s1.insert(9); 
    s1.insert(11); 
    set<int>::value_compare myComp; 
    myComp = s1.value_comp(); 
    set<int>::iterator iter1 = s1.find(45); 
    set<int>::iterator iter2 = s1.find(9); 
    cout << myComp(*iter1, *iter2)<<endl; 
    return 0;
}
